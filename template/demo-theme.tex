\titleframe

\begin{frame}{What this demo covers}
  \begin{itemize}
    \item Frame titles with a section eyebrow and a hairline rule.
    \item Thin-rule blocks, inline \alert{semantic emphasis}, and theorems.
    \item Tables, figures with numbered captions, and a bibliography.
    \item Section dividers, special frames, and the quiet footer.
  \end{itemize}
\end{frame}

\section{Blocks and semantics}

\begin{frame}{Blocks carry meaning by color, not decoration}
  \begin{block}{Claim}
    Blocks are marked by a thin colored left rule and a faint tinted body.
  \end{block}
  \begin{alertblock}{Caveat}
    Alert content is gold, matching inline \alert{alerted text}.
  \end{alertblock}
  \begin{exampleblock}{Desired behavior}
    Example content is green.
  \end{exampleblock}
\end{frame}

\begin{frame}{Theorems and math}
  \begin{theorem}[Selection rule]
    Include a figure only when its evidence value is high:
    \[ U(f) = \alpha R(f) + \beta E(f) - \gamma C(f). \]
  \end{theorem}
  \begin{definition}
    \emph{Evidence value} is the information a figure adds beyond its caption.
  \end{definition}
\end{frame}

\begin{frame}{Lists, descriptions, and references}
  \begin{columns}[T,totalwidth=\textwidth]

      \end{thebibliography}
    \end{column}
  \end{columns}
\end{frame}
